databricks/etl_antidepressivos.py
Python · PySpark
# Importa as funções necessárias do PySpark para transformação dos dados
from pyspark.sql.functions import col, upper, trim, round, when

# Carrega a tabela de dados brutos armazenada no ambiente de dados
dados_brutos = spark.table("workspace.default.antidepressivos_brasil_unificado")

# Inicia o processo de limpeza e padronização dos dados
tabela_limpa = (
    dados_brutos
    .select(
        # ==========================
        # IDENTIFICAÇÃO E PERÍODO
        # ==========================

        # Converte o identificador para inteiro
        col("id_linha").cast("int").alias("id_linha"),

        # Converte o ano para inteiro e renomeia a coluna
        col("ano").cast("int").alias("ano_referencia"),

        # ==========================
        # PADRONIZAÇÃO DE TEXTO
        # ==========================

        # Remove espaços extras e converte o nome da região para letras maiúsculas
        upper(trim(col("regiao"))).alias("regiao_nome"),

        # Remove espaços extras e converte o nome do estado para letras maiúsculas
        upper(trim(col("estado"))).alias("estado_nome"),

        # ==========================
        # TRATAMENTO DE DADOS NULOS
        # ==========================

        # Caso a UF esteja nula, atribui "BR" para representar dados nacionais
        # Caso contrário, converte a sigla para letras maiúsculas
        when(col("uf").isNull(), "BR")
        .otherwise(upper(col("uf")))
        .alias("uf_sigla"),

        # ==========================
        # MÉTRICAS DE VENDAS
        # ==========================

        # Arredonda a quantidade estimada de unidades vendidas
        # e converte para número inteiro longo
        round(col("unidades_vendidas_est"), 0)
        .cast("long")
        .alias("vendas_unidades"),

        # ==========================
        # INDICADORES DE SAÚDE
        # ==========================

        # Arredonda a prevalência de depressão para duas casas decimais
        round(col("prev_depressao_regiao_pct"), 2)
        .alias("percentual_prevalencia_depressao"),

        # Arredonda a densidade de psiquiatras para duas casas decimais
        round(col("psiquiatras_por_100mil"), 2)
        .alias("densidade_psiquiatras_por_100mil"),

        # ==========================
        # KPI ANALÍTICO
        # ==========================

        # Calcula a quantidade de unidades vendidas para cada 100 mil habitantes
        # permitir comparações proporcionais entre estados e regiões
        round(
            (col("unidades_vendidas_est") / col("populacao_2022")) * 100000,
            2
        ).alias("vendas_por_100mil_hab")
    )
)

# Salva o resultado final como uma tabela Delta otimizada para o Power BI
tabela_limpa.write \
    .format("delta") \
    .mode("overwrite") \
    .option("overwriteSchema", "true") \
    .saveAsTable("workspace.default.antidepressivos_vendas_limpo")

# Exibe uma amostra dos dados tratados
display(tabela_limpa.limit(10))
databricks/regressao_linear.py
Python · Scikit-Learn
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
from sklearn.metrics import r2_score, mean_absolute_error

# Carrega os dados tratados do Spark para o Pandas (ambiente Databricks)
df = spark.table("workspace.default.antidepressivos_vendas_limpo").toPandas()

# Agrupa as vendas totais por ano para análise nacional
vendas_anuais = df.groupby("ano_referencia")["vendas_unidades"].sum().reset_index()

# Prepara as variáveis para o modelo (X = Ano, y = Vendas)
X = vendas_anuais[["ano_referencia"]].values
y = vendas_anuais["vendas_unidades"].values

# Inicializa e treina o modelo de Regressão Linear
modelo = LinearRegression()
modelo.fit(X, y)

# Realiza as previsões baseadas nos anos históricos
previsoes = modelo.predict(X)

# Calcula métricas de desempenho do modelo
r2 = r2_score(y, previsoes)
mae = mean_absolute_error(y, previsoes)

print(f"Coeficiente de Determinação (R²): {r2:.4f}")
print(f"Erro Médio Absoluto (MAE): {mae:.2f}")

# Projeção para os próximos 3 anos (2025, 2026, 2027)
anos_futuros = np.array([[2025], [2026], [2027]])
projecao_vendas = modelo.predict(anos_futuros)

# Exibe os resultados da projeção
for ano, valor in zip(anos_futuros.flatten(), projecao_vendas):
    print(f"Projeção de vendas para {ano}: {valor:,.0f} unidades")

# Exporta os resultados da projeção para uma tabela Spark
df_projecao = pd.DataFrame({
    "ano": anos_futuros.flatten(),
    "vendas_projetadas": projecao_vendas
})

spark.createDataFrame(df_projecao).write \
    .mode("overwrite") \
    .saveAsTable("workspace.default.antidepressivos_projecao_linear")
databricks/previsao_prophet.py
Python · Prophet
from prophet import Prophet
import pandas as pd

# Carrega dados históricos consolidados
df_spark = spark.table("workspace.default.antidepressivos_vendas_limpo")
df_pandas = df_spark.toPandas()

# Prepara o DataFrame para o padrão exigido pelo Prophet
# ds: coluna de data (ano), y: coluna de valor (vendas)
df_prophet = (
    df_pandas.groupby("ano_referencia")["vendas_unidades"]
    .sum()
    .reset_index()
    .rename(columns={"ano_referencia": "ds", "vendas_unidades": "y"})
)

# Converte o ano para o primeiro dia de janeiro para o Prophet interpretar como data
df_prophet["ds"] = pd.to_datetime(df_prophet["ds"].astype(str) + "-01-01")

# Inicializa o modelo Prophet com intervalo de confiança de 95%
modelo = Prophet(interval_width=0.95, yearly_seasonality=True)

# Treina o modelo com os dados históricos
modelo.fit(df_prophet)

# Cria um DataFrame com os próximos 5 anos para previsão
futuro = modelo.make_future_dataframe(periods=5, freq='Y')

# Gera as previsões
previsao = modelo.predict(futuro)

# Seleciona as colunas de interesse para exportação
forecast_out = previsao[
    [
        "ds",      # Data
        "yhat",    # Previsão Central
        "yhat_lower", # Limite Inferior
        "yhat_upper"  # Limite Superior
    ]
].copy()

# Renomeia as colunas para facilitar consumo no Power BI
forecast_out.columns = [
    "data_referencia",
    "previsao",
    "ic_inferior_95",
    "ic_superior_95"
]

# Salva a previsão em uma tabela Delta
spark.createDataFrame(forecast_out).write \
    .format("delta") \
    .mode("overwrite") \
    .saveAsTable(
        "workspace.default.antidepressivos_previsao_prophet"
    )

# Mensagem de confirmação
print(
    "Tabela salva: antidepressivos_previsao_prophet"
)
